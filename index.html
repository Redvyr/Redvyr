Real Script

KillStreak system — live BillboardGui above players' heads, resets on death.

KillStreakSTATS.lua
local Players = game:GetService("Players") -- Create BillboardGui above player head local function createBillboardGui(character) if character:FindFirstChild("Head") then local gui = Instance.new("BillboardGui") gui.Size = UDim2.new(4,0,2,0) gui.StudsOffset = Vector3.new(0,3,0) gui.AlwaysOnTop = true local label = Instance.new("TextLabel") label.Text = "0" label.Parent = gui gui.Parent = character.Head end end -- Reset on death, increment on kill local function onDied(player) player.leaderstats.KillStreak.Value = 0 end local function onKill(victim, killer) if killer ~= victim then killer.leaderstats.KillStreak.Value += 1 end end Players.PlayerAdded:Connect(onPlayerAdded)
